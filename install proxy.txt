На новый сервер нужно перенести конфиг 
/etc/nginx/nginx.conf
и скрипт
/usr/local/bin/cf-ips-update.sh

команды:
apt -y update
apt -y install nginx
echo '03 05 * * * root /usr/local/bin/cf-ips-update.sh 1>/dev/null 2>/dev/null' > /etc/cron.d/cf-ips
/usr/local/bin/cf-ips-update.sh
